procedure binaryInsertionSort (a[1], a[2], ..., a[n])
//a[1], a[2], ..., a[n]: các số thực với n >= 2
for i:=2 to n
	j := i - 1
	k := 1
	selected := a
	while k < j
		m := (j + k) / 2
		if selected > a[m] then k := m + 1
		else j := m
	if selected >= a[k] then location := k + 1
	else location := k
	for v := i - 1 to location (với điều kiện chạy vong lập là v >= location)
		a[v+1] = a[v]
	a[location] = selected
